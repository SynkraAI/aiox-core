**@remotion-architect + @motion-master:**

```typescript
// liquid-morph-text.tsx
import { interpolate, useCurrentFrame, Easing } from "remotion";

export const LiquidMorphText: React.FC<{
  text: string;
  startFrame: number;
  endFrame: number;
}> = ({ text, startFrame, endFrame }) => {
  const frame = useCurrentFrame();

  // Liquid entrance
  const scaleY = interpolate(
    frame,
    [startFrame, startFrame + 15],
    [0, 1.2, 1],
    {
      extrapolateRight: "clamp",
      easing: Easing.bezier(0.68, -0.55, 0.27, 1.55), // Elastic
    }
  );

  const blur = interpolate(
    frame,
    [startFrame, startFrame + 8],
    [20, 0],
    { extrapolateRight: "clamp" }
  );

  return (
    <div
      style={{
        fontFamily: "Space Grotesk",
        fontSize: 96,
        fontWeight: 700,
        color: "#FFFFFF",
        transform: `scaleY(${scaleY})`,
        filter: `blur(${blur}px)`,
        textAlign: "left", // v5.0: left-aligned
        paddingLeft: 60, // v5.0: asymmetric
      }}
    >
      {text}
    </div>
  );
};

// gradient-overlay.tsx
export const GradientOverlay: React.FC<{
  opacity?: number;
}> = ({ opacity = 0.15 }) => {
  return (
    <div
      style={{
        position: "absolute",
        top: 0,
        left: 0,
        width: "100%",
        height: "100%",
        background: "linear-gradient(135deg, #C9B298 0%, #D4C4AB 100%)",
        opacity,
        mixBlendMode: "overlay",
        pointerEvents: "none",
      }}
    />
  );
};

// fake-3d-layer.tsx
export const Fake3DLayer: React.FC<{
  children: React.ReactNode;
  depth: number; // 1-5
}> = ({ children, depth }) => {
  const frame = useCurrentFrame();

  const rotateX = interpolate(
    frame,
    [0, 120],
    [0, 5],
    { extrapolateRight: "clamp" }
  );

  const translateZ = depth * 20; // Fake depth

  return (
    <div
      style={{
        transform: `
          perspective(1000px)
          rotateX(${rotateX}deg)
          translateZ(${translateZ}px)
        `,
        transformStyle: "preserve-3d",
      }}
    >
      {children}
    </div>
  );
};
```
